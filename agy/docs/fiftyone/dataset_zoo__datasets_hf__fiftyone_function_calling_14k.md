# FiftyOne Function Calling 14k Dataset#

## Overview#

This dataset is derived from the FiftyOne documentation and is designed to train AI assistants to understand and answer questions about FiftyOneâs functionality. The dataset follows the format specified in the APIGen paper, structuring the data to map natural language queries to appropriate API tools and their usage.

## Purpose#

  * Train AI models to understand FiftyOne-related queries

  * Provide structured examples of FiftyOne tool usage

  * Map natural language questions to appropriate FiftyOne methods

  * Demonstrate correct parameter usage for FiftyOne functions




## Dataset Structure (APIGen Format)#

Following the [APIGen paper](https://arxiv.org/pdf/2406.18518) (Section A.2.1), the dataset is structured as a JSON object with three main components:

### Components#

  1. **query** (array of strings)

     * Each string represents a problem statement or question

     * Natural language queries about FiftyOne functionality

  2. **tools** (array of JSON strings)

     * Each tool object contains:

       * `name` (string): The name of the tool

       * `description` (string): Brief description of what the tool does

       * `parameters` (object): Required and optional parameters where each parameter has:

         * `type` (string): Data type (e.g., âintegerâ, âfloatâ, âarrayâ)

         * `description` (string): Brief description of the parameter

         * `required` (boolean): Whether the parameter is mandatory

  3. **answers** (array of JSON strings)

     * Each answer object contains:

       * `name` (string): The tool used to generate the answer

       * `arguments` (object): Key-value pairs of parameters and their values




## Example Entry#

      "query": [

The code used to generate this dataset was:
    
    
    import os
    import json
    import time
    import re
    from datetime import datetime
    from pathlib import Path
    from typing import List, Dict, Any, Optional
    
    
    import anthropic
    import instructor
    from pydantic import BaseModel, Field
    
    class FiftyOneQuery(BaseModel):
        """Model representing a query about FiftyOne."""
        query: List[str] = Field(description="List of clear questions or problems about FiftyOne")
    
    class ToolParameter(BaseModel):
        """Model representing a parameter for a FiftyOne tool."""
        type: str = Field(description="The exact data type (string, integer, float, array, etc.)")
        description: str = Field(description="Clear explanation of the parameter")
        required: bool = Field(description="Boolean indicating if parameter is mandatory")
    
    class FiftyOneTool(BaseModel):
        """Model representing a FiftyOne tool/method."""
        name: str = Field(description="The exact method name")
        description: str = Field(description="Clear explanation of what the tool does")
        parameters: Dict[str, ToolParameter] = Field(description="Object containing all parameters")
    
    class ToolArgument(BaseModel):
        """Model representing arguments for a FiftyOne tool call."""
        name: str = Field(description="The exact FiftyOne function or class method used")
        arguments: Dict[str, Any] = Field(description="Object containing the actual values passed to the tool's parameters")
    
    class FiftyOneExamples(BaseModel):
        """Model representing FiftyOne examples with queries, tools, and answers."""
        query: List[str] = Field(description="List of clear questions or problems about FiftyOne")
        tools: List[str] = Field(description="List of JSON strings containing arrays of tool definitions")
        answers: List[str] = Field(description="List of JSON strings containing arrays of tool calls")
    
    class APIProcessor:
        def __init__(self, api_key, model, output_dir="output"):
            self.model = model
            self.output_dir = Path(output_dir)
            self.output_dir.mkdir(exist_ok=True)
            
            # Set up Anthropic client with instructor
            self.client = instructor.from_anthropic(
                anthropic.Anthropic(api_key=api_key),
                mode=instructor.Mode.ANTHROPIC_TOOLS
            )
            
            # Single files that we'll keep appending to
            self.raw_log_path = self.output_dir / "raw_responses.jsonl"
            self.processed_path = self.output_dir / "processed_responses.json"
            self.error_log_path = self.output_dir / "errors.log"
            
            self.processed_data = self.load_processed_data()
    
        def load_processed_data(self):
            if self.processed_path.exists():
                print(f"Loading existing processed data from {self.processed_path}")
                with open(self.processed_path, 'r') as f:
                    existing_data = json.load(f)
                    print(f"Found {len(existing_data['query'])} existing examples")
                    return existing_data
            print("No existing processed data found, starting fresh")
            return {
                "query": [],
                "tools": [],
                "answers": []
            }
    
        def get_markdown_files(self, directory, limit=None):
            markdown_files = []
            for root, _, files in os.walk(directory):
                for file in files:
                    if file.endswith('.md'):
                        markdown_files.append(Path(root) / file)
                        if limit and len(markdown_files) >= limit:
                            return markdown_files
            return markdown_files
    
        def clean_markdown_content(self, content):
            """Remove base64 encoded images and HTML from markdown content"""
            # Store original length for logging
            original_length = len(content)
            
            # Patterns for base64 images
            base64_pattern = r'!\[.*?\]\(data:image\/[^;]+;base64,[^)]+\)'
            html_base64_pattern = r'<img[^>]+src="data:image\/[^;]+;base64,[^"]+"[^>]*>'
            
            # Common HTML patterns that should be removed with their content
            remove_with_content = {
                'comments': r'<!--[\s\S]*?-->',
                'scripts': r'<script[\s\S]*?</script>',
                'styles': r'<style[\s\S]*?</style>',
            }
            
            # Remove base64 images first
            cleaned = re.sub(base64_pattern, '[IMAGE]', content)
            cleaned = re.sub(html_base64_pattern, '[IMAGE]', cleaned)
            
            # Remove elements with their content
            for element_type, pattern in remove_with_content.items():
                cleaned = re.sub(pattern, '', cleaned)
            
            # Clean markdown links that contain code references
            # This pattern looks for [`something`](https://docs.voxel51.com/dataset_zoo/datasets_hf/link "title") and preserves just the `something`
            cleaned = re.sub(r'\[(`[^`]+`)\]\([^)]+\s+"[^"]+"\)', r'\1', cleaned)
            
            # Remove just the HTML tags, preserve content
            cleaned = re.sub(r'<[^>]*>', '', cleaned)
            
            # Clean up excessive newlines
            cleaned = re.sub(r'\n\s*\n\s*\n+', '\n\n', cleaned)
            
            return cleaned
    
        def split_content_into_chunks(self, content):
            """Split content into chunks if word count exceeds threshold"""
            # Split into words while preserving newlines
            words = content.split()
            total_words = len(words)
            
            if total_words > 2000:
                chunk_size = total_words // 5
            else:
                # If under threshold, keep as single chunk
                return [content]
            
            # Create three chunks
            chunks = []
            current_pos = 0
            
            for i in range(5):
                if i == 2:  # Last chunk gets any remaining words
                    chunk_words = words[current_pos:]
                else:
                    chunk_words = words[current_pos:current_pos + chunk_size]
                
                # Rejoin words with spaces and preserve paragraph structure
                chunk = ' '.join(chunk_words)
                chunks.append(chunk)
                current_pos += chunk_size
            
            print(f"Split content into 5 chunks of approximately {chunk_size} words each")
            return chunks
    
        def process_file(self, file_path, prompt):
            print(f"Processing: {file_path}")
            
            try:
                with open(file_path, "r", encoding="utf-8") as f:
                    content = f.read()
                
                # Clean the content before splitting
                cleaned_content = self.clean_markdown_content(content)
                
                # Split into chunks
                chunks = self.split_content_into_chunks(cleaned_content)
                
                all_parsed_data = {
                    "query": [],
                    "tools": [],
                    "answers": []
                }
                
                # Process each third
                for i, chunk in enumerate(chunks, 1):
                    print(f"Processing chunk {i}/5")
                    
                    try:
                        # Use instructor to directly get the structured data
                        examples = self.client.chat.completions.create(
                            model=self.model,
                            max_tokens=8192,
                            messages=[
                                {
                                    "role": "system",
                                    "content": prompt
                                },
                                {
                                    "role": "user",
                                    "content": f"Use the following content to generate the structured FiftyOne examples as requested. This is part {i} of 3 from the document:\n\n{chunk}"
                                }
                            ],
                            response_model=FiftyOneExamples,
                        )
                        
                        # Log raw response for debugging purposes
                        log_entry = {
                            "timestamp": datetime.now().isoformat(),
                            "file": str(file_path),
                            "chunk": i,
                            "total_chunks": 5,
                            "response": json.dumps(examples.model_dump())
                        }
                        with open(self.raw_log_path, 'a') as f:
                            f.write(json.dumps(log_entry) + '\n')
                        
                        # Accumulate results from this chunk
                        all_parsed_data["query"].extend(examples.query)
                        all_parsed_data["tools"].extend(examples.tools)
                        all_parsed_data["answers"].extend(examples.answers)
                        
                    except instructor.exceptions.IncompleteOutputException as e:
                        print(f"Validation error processing chunk {i}: {e}")
                        with open(self.error_log_path, 'a') as f:
                            f.write(f"{datetime.now().isoformat()}: Validation error processing chunk {i} of {file_path}: {str(e)}\n")
                        continue
    
                # Update processed data with accumulated results from all chunks
                self.processed_data["query"].extend(all_parsed_data["query"])
                self.processed_data["tools"].extend(all_parsed_data["tools"])
                self.processed_data["answers"].extend(all_parsed_data["answers"])
    
                # Save updated processed data
                with open(self.processed_path, 'w') as f:
                    json.dump(self.processed_data, f, indent=2)
    
                print(f"Successfully processed all chunks for: {file_path}")
    
            except Exception as e:
                print(f"Error processing {file_path}: {str(e)}")
                with open(self.error_log_path, 'a') as f:
                    f.write(f"{datetime.now().isoformat()}: Error processing {file_path}: {str(e)}\n")
                    
        def process_directory(self, directory, prompt, test_files=None):
            markdown_files = self.get_markdown_files(directory, limit=test_files)
            print(f"Found {len(markdown_files)} markdown files to process")
            
            for file_path in markdown_files:
                self.process_file(file_path, prompt)
